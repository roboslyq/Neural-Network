class PoetryGenerator:
    def __init__(self):
        # 初始化模型组件
        self.tokenizer = ChineseTokenizer()
        self.embedding = nn.Embedding(vocab_size, 768)
        self.transformer_layers = [TransformerLayer() for _ in range(12)]
        self.kv_cache = {}  # 用于存储各层的键值对

    def encode_prompt(self, prompt: str):
        """处理用户输入的提示词"""
        # 分词和向量化
        prompt_tokens = self.tokenizer.encode(prompt)
        hidden_states = self.embedding(prompt_tokens)
        
        # 构建初始KV缓存
        for layer_idx, layer in enumerate(self.transformer_layers):
            k, v = layer.compute_kv(hidden_states)
            self.kv_cache[layer_idx] = (k, v)
        
        return prompt_tokens

    def generate_poem(self, prompt: str, max_length=28):
        """主生成函数"""
        # 1. 编码提示词
        prompt_tokens = self.encode_prompt(prompt)
        
        # 2. 初始化生成状态
        generated = [BOS_TOKEN]
        current_pos = 0
        rhyme_scheme = []  # 押韵模式记录
        
        while len(generated) < max_length:
            # 3. 获取当前token的向量表示
            token_embed = self.embedding([generated[-1]])
            
            # 4. 逐层处理（展示单层逻辑）
            for layer_idx, layer in enumerate(self.transformer_layers):
                # 从缓存获取当前层的键值对
                k_prompt, v_prompt = self.kv_cache[layer_idx]
                
                # 计算跨注意力（当前token vs 提示词）
                query = layer.attention.query(token_embed)
                attn_scores = torch.matmul(query, k_prompt.transpose(-1, -2))
                attn_weights = softmax(attn_scores / sqrt(d_model))
                
                # 重点关注"写"和"七言绝句"相关位置
                if layer_idx == 6:  # 中间层强化文体特征
                    attn_weights[:, :, 3:5] *= 1.5  # 提升权重
                
                # 值向量加权求和
                context = torch.matmul(attn_weights, v_prompt)
                
                # 更新隐藏状态（残差连接）
                token_embed = layer.mlp(context + token_embed)
            
            # 5. 预测下一个token的概率分布
            logits = self.output_layer(token_embed)
            
            # 6. 应用古诗约束
            if should_apply_rhyme(len(generated)):
                logits = adjust_rhyme(logits, rhyme_scheme)
            if should_limit_length(len(generated)):
                logits = forbid_extra_length(logits)
            
            # 7. 采样下一个token
            next_token = sample_with_temperature(logits, temp=0.7)
            
            # 8. 更新生成状态
            generated.append(next_token)
            update_rhyme_scheme(next_token, rhyme_scheme)
            
            # 9. 更新KV缓存
            new_k, new_v = self.transformer_layers[0].compute_kv(token_embed)
            for layer in self.transformer_layers:
                layer.kv_cache = append_new_kv(layer.kv_cache, new_k, new_v)
        
        return decode_tokens(generated[1:-1])  # 去除BOS/EOS
        
        # 辅助函数示例
def adjust_rhyme(logits, rhyme_scheme):
    """在押韵位置调整概率分布"""
    if len(rhyme_scheme) < 3:
        return logits
    last_rhyme = get_rhyme(rhyme_scheme[-1])
    rhyme_mask = create_rhyme_mask(last_rhyme) 
    return logits + rhyme_mask * 2.0  # 押韵字概率提升

def should_apply_rhyme(current_pos):
    """判断是否在押韵位置（每句末字）"""
    return (current_pos + 1) % 7 == 0

# 使用示例
generator = PoetryGenerator()
poem = generator.generate_poem("请帮忙我写一首七言绝句")

大模型工作原理：

循环生成步骤中，大模型会将已生成的内容作为新的输入，回到向量表示阶段，进行下一个字的生成。

这种自回归机制是大模型生成连贯文本的关键。每个新生成的字都会影响后续生成的内容，创造出连贯的文本序列。
